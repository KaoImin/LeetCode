# 766 [托普利茨矩阵](https://leetcode-cn.com/problems/toeplitz-matrix/)

```rust
impl Solution {
    pub fn is_toeplitz_matrix(matrix: Vec<Vec<i32>>) -> bool {
        for i in 0..matrix.len() - 1 {
            let mut line1 = matrix[i].clone();
            let mut line2 = matrix[i + 1].clone();
            line1.pop();
            line2.remove(0);
            
            if line1 != line2 {
                return false;
            }
        }
        true
    }
}
```
